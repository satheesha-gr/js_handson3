Find the Fives.

        const Find_Five = (N) => 
        {
	        let count=0;
	        while (N > 0) {
	            if (N%10 === 5) {
	                count++;
	            }
	            N = Math.floor(N/10);
	        }
	        return count;
        };

Find Sum

        const findSum = (n) => {
            let sum=0;
            for (let i=0;i<=n;i+=2){
                sum = sum + i;
            }
            return sum;
        };

Find the sum of the digits of a given number.

        const Number_Sum = (N) => 
        {
	        let sum=0;
	        while (N > 0) {
	            sum += N%10;
	            N=Math.floor(N/10);
	        }

Check whether a Number is a prime or not.

        const Prime_Check = (N) => 
        {
            if (N <= 1) {
                return "NO";
            }
            for (let i=2; i<=Math.sqrt(N); i++) {
                if (N%i === 0) {
                    return "NO";
                }
            }
            return "YES";
        };
